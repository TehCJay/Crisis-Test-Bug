# Haughty turns into Proud, Arbitrary or Cruel or RUTHLESS
character_event = {
	id = ZE.13000

	hide_window = yes
	
	is_triggered_only = yes
	
	trigger = {
		trait = haughty
	}
	
	immediate = {
		if = {
			limit = {				
				trait = incapable
				age = 15
			}
			remove_trait = haughty
			break = yes
		}
		if = {
			limit = {				
				personality_traits = 5
			}
			remove_trait = haughty
			break = yes
		}
		set_character_flag = was_haughty
		random_list = {  # Select outcome
			100 = {
				set_character_flag = haughty_to_proud
				modifier = {
					factor = 0
					OR = {
						trait = humble
						trait = proud
					}
				}
				modifier = {
					factor = 2
					educator = {
						trait = proud
					}
				}
			}
			100 = {
				set_character_flag = haughty_to_arbitrary
				modifier = {
					factor = 0
					OR = {
						trait = just
						trait = arbitrary
					}
				}
				modifier = {
					factor = 2
					educator = {
						trait = arbitrary
					}
				}
			}
			100 = {
				set_character_flag = haughty_to_cruel
				modifier = {
					factor = 0
					OR = {
						trait = kind
						trait = cruel
					}
				}
				modifier = {
					factor = 2
					educator = {
						trait = cruel
					}
				}
			}
			100 = {
				set_character_flag = haughty_to_ruthless
				modifier = {
					factor = 0
					OR = {
						trait = ruthless
						trait = honorable
					}
				}
				modifier = {
					factor = 2
					educator = {
						trait = ruthless
					}
				}
			}
			1 = { #None found
				modifier = {
					factor = 0.01
				}
				clr_character_flag = was_haughty
				remove_trait = haughty
				break = yes
			}
		}
		if = {
			limit = {
				NOT = { trait = ambitious }
				NOT = { trait = content }
				educator = { trait = ambitious }
				prisoner = no
			}
			educator = { character_event = { id = ZE.13002 } }  # Let educator decide on intervention
			break = yes
		}
		character_event = { id = ZE.13001 } # Go direct to trait change
	}
}
