				FROM = { trait = proud }
			}

				factor = 0
				FROM = { trait = disciplined }

			}

				factor = 0

			}

				factor = 0
				tier = emperor
				NOT = { reverse_has_opinion_modifier = { who = FROM modifier = opinion_owes_me_a_favor } }
				FROM = { tier = duke }
			}

				factor = 2
